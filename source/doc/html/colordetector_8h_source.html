63  // Computes the city-block distance between two colors.
64  int getColorDistance(const cv::Vec3b& color1, const cv::Vec3b& color2) const {

66  return abs(color1[0]-color2[0])+

79  // Processes the image. Returns a 1-channel binary image.

82  cv::Mat operator()(const cv::Mat &image) {
83 
84  cv::Mat input;
85 
86  if (useLab) { // Lab conversion
87  cv::cvtColor(image, input, CV_BGR2Lab);
88  }
89  else {
90  input = image;
91  }
92 
93  cv::Mat output;
94  // compute absolute difference with target color
95  cv::absdiff(input,cv::Scalar(target),output);
96  // split the channels into 3 images
97  std::vector<cv::Mat> images;
98  cv::split(output,images);
99  // add the 3 channels (saturation might occurs here)
100  output= images[0]+images[1]+images[2];
101  // apply threshold
102  cv::threshold(output, // input image
103  output, // output image
104  maxDist, // threshold (must be < 256)
105  255, // max value
106  cv::THRESH_BINARY_INV); // thresholding type
107 
108  return output;
109  }

144  target = tmp.at<cv::Vec3b>(0, 0);
145  }
